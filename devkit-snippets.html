Быстрая сортировка (QuickSort) C++
void quicksort(vector<int>& a, int l, int r) { if (l >= r) return; int x = a[(l + r) / 2]; int i = l, j = r; while (i <= j) { while (a[i] < x) i++; while (a[j] > x) j--; if (i <= j) swap(a[i++], a[j--]); } quicksort(a, l, j); quicksort(a, i, r); }
Бинарный поиск C++
int binary_search(vector<int>& a, int target) { int l = 0, r = a.size() - 1; while (l <= r) { int m = l + (r - l) / 2; if (a[m] == target) return m; if (a[m] < target) l = m + 1; else r = m - 1; } return -1; }
DFS (поиск в глубину) C++
vector<vector<int>> g; vector<bool> used; void dfs(int v) { used[v] = true; for (int to : g[v]) { if (!used[to]) dfs(to); } }
BFS (поиск в ширину) C++
vector<vector<int>> g; vector<int> dist; void bfs(int start) { queue<int> q; q.push(start); dist[start] = 0; while (!q.empty()) { int v = q.front(); q.pop(); for (int to : g[v]) { if (dist[to] == -1) { dist[to] = dist[v] + 1; q.push(to); } } } }
ДП: Лестница C++
int climbStairs(int n) { if (n <= 2) return n; int prev1 = 1, prev2 = 2; for (int i = 3; i <= n; i++) { int cur = prev1 + prev2; prev1 = prev2; prev2 = cur; } return prev2; }
Алгоритм Дейкстры C++
vector<int> dijkstra(vector<vector<pair<int,int>>>& g, int s) { vector<int> dist(g.size(), INF); dist[s] = 0; priority_queue<pair<int,int>> pq; pq.push({0, s}); while (!pq.empty()) { int d = -pq.top().first, v = pq.top().second; pq.pop(); if (d > dist[v]) continue; for (auto [to, w] : g[v]) { if (dist[v] + w < dist[to]) { dist[to] = dist[v] + w; pq.push({-dist[to], to}); } } } return dist; }
