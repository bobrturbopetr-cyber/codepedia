## Пример кода

```python
# Hello World
print("Hello, Codepedia!")

# Функция
def greet(name: str) -> str:
    return f"Hello, {name}!"

# Класс
class Developer:
    def __init__(self, name, language):
        self.name = name
        self.language = language
    
    def code(self):
        print(f"{self.name} пишет на {self.language}")

# Использование
dev = Developer("Анна", "Python")
dev.code()
