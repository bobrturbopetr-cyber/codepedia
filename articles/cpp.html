C++ — компилируемый статически типизированный язык программирования общего назначения. Создан Бьёрном Страуструпом в 1985 году как расширение языка C. Даёт полный контроль над аппаратным обеспечением.

⚙️ Ключевые особенности

  • Компилируемый — код превращается в машинный, максимальная производительность
  • Статическая типизация — типы проверяются на этапе компиляции
  • Прямой доступ к памяти — указатели, ручное управление
  • Zero-cost абстракции — абстракции не замедляют код
  • RAII — автоматическое управление ресурсами через конструкторы/деструкторы
  • Мультипарадигмальность — ООП, обобщённое, функциональное программирование

💻 Пример кода (современный C++17)

#include <iostream>
#include <vector>
#include <memory>
#include <optional>
#include <algorithm>

// Функция с std::optional
std::optional<int> safe_divide(int a, int b) {
    if (b == 0) return std::nullopt;
    return a / b;
}

// Класс с RAII
template<typename T>
class SafeArray {
private:
    std::unique_ptr<T[]> data;
    size_t size_;
    
public:
    explicit SafeArray(size_t size) 
        : data(std::make_unique<T[]>(size)), size_(size) {
        std::cout << "Создан массив из " << size << " элементов\n";
    }
    
    T& operator[](size_t index) {
        if (index >= size_) throw std::out_of_range("Индекс за пределами");
        return data[index];
    }
    
    size_t size() const { return size_; }
};

// Lambda-выражение
auto square = [](int x) { return x * x; };

int main() {
    std::cout << "🚀 Современный C++\n";
    
    // Умные указатели (вместо new/delete)
    auto ptr = std::make_unique<int>(42);
    std::cout << "*ptr = " << *ptr << '\n';
    
    // STL контейнеры и алгоритмы
    std::vector<int> numbers = {1, 2, 3, 4, 5};
    std::vector<int> squares;
    std::transform(numbers.begin(), numbers.end(), 
                   std::back_inserter(squares), square);
    
    for (int n : squares) {
        std::cout << n << " ";
    }
    std::cout << '\n';
    
    // SafeArray с RAII
    SafeArray<int> arr(5);
    for (int i = 0; i < arr.size(); ++i) {
        arr[i] = i * i;
    }
    
    // Безопасное деление
    if (auto result = safe_divide(10, 2)) {
        std::cout << "10 / 2 = " << *result << '\n';
    }
    
    return 0;
}

🗃️ Управление памятью в C++

// ❌ ПЛОХО: ручное управление (старый стиль)
int* raw_ptr = new int(10);
delete raw_ptr;  // Если забудете — утечка!

// ✅ ХОРОШО: умные указатели (современный стиль)
auto smart = std::make_unique<int>(10);
// удалится автоматически при выходе из области видимости

// 🔄 Идиома RAII для любых ресурсов
class FileHandle {
    FILE* f;
public:
    FileHandle(const char* name) : f(fopen(name, "r")) {}
    ~FileHandle() { if (f) fclose(f); }  // Автоматическое закрытие
    // Запрещаем копирование
    FileHandle(const FileHandle&) = delete;
    FileHandle& operator=(const FileHandle&) = delete;
};
